96 void PhysicsSystem::Solve(std::vector<std::shared_ptr<Entity>>& entities, std::vector<std::shared_ptr<Collision>>& collisions, std::unordered_map<int, int>& idToIndexMap)
97 {
98  float ELASTICITY = 1.f;
99  for (int i = 0; i < collisions.size(); i++)
100  {
101  std::shared_ptr<Collision> collision = collisions[i];
102  int firstEntityIndex = idToIndexMap[collision->first];
103  int secondEntityIndex = idToIndexMap[collision->second];
104 
105  PhysicsComponent* first = entities[firstEntityIndex]->GetComponent<PhysicsComponent>(ComponentType::Physics);
106  PhysicsComponent* second = entities[secondEntityIndex]->GetComponent<PhysicsComponent>(ComponentType::Physics);
107 
108  std::shared_ptr<Collider> firstCollider = collision->firstCollider;
109  std::shared_ptr<Collider> secondCollider = collision->secondCollider;
110 
111  for (int j = 0; j < collision->contacts.size(); j++)
112  {
113  Contact contact = collision->contacts[j];
114  glm::vec3 normal = contact.contactNormal;
115  glm::vec3 rA = contact.contactPoint - firstCollider->center;
116  glm::vec3 rB = contact.contactPoint - secondCollider->center;
117  glm::vec3 vA = first->velocity;
118  glm::vec3 wA = first->angularVel;
119  glm::vec3 vB = second->velocity;
120  glm::vec3 wB = second->angularVel;
121  float invMassA = first->inverseMass;
122  float invMassB = second->inverseMass;
123  glm::mat3 invInertiaTensorA = first->invInertiaTensor;
124  glm::mat3 invInertiaTensorB = second->invInertiaTensor;
125 
126  /*
127  j = nominator / denominator
128  nominator = -(1 + e) * n * ( (vA + wA x rA) - (vB + wB x rB) )
129  denominator = 1/mA + 1/mB + n * ( (invIA (rA x n) x rA) + (invIB (rB x n) x rB) )
130  ----------- --------------------- ---------------------
131  / / /
132  / / /
133  invMassSum angularPart1 angularPart2
134  */
135  glm::vec3 preRelVelocity = vA + glm::cross(wA, rA) - vB - glm::cross(wB, rB);
136  float nominator = glm::dot((-1.0f - ELASTICITY) * preRelVelocity, normal);
137  float invMassSum = invMassA + invMassB;
138  glm::vec3 angularPart1 = glm::cross(invInertiaTensorA * glm::cross(rA, normal), rA);
139  glm::vec3 angularPart2 = glm::cross(invInertiaTensorB * glm::cross(rB, normal), rB);
140  float angularFinal = glm::dot(angularPart1 + angularPart2, normal);
141 
142  float impulse = nominator / (invMassSum + angularFinal);
143 
144  if (firstCollider->dynamicType != DynamicType::Static)
145  {
146  first->velocity = vA + impulse * normal * invMassA;
147  first->angularVel = wA + invInertiaTensorA * glm::cross(rA, impulse * normal);
148  }
149  if (secondCollider->dynamicType != DynamicType::Static)
150  {
151  second->velocity = vB - impulse * normal * invMassB;
152  second->angularVel = wB - invInertiaTensorB * glm::cross(rB, impulse * normal);
153  }
154  }
155  }
156 }
